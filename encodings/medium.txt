% Sistema di selezione dinamica delle strategie in base alla memoria dei fatti precedenti

% ===================================
% DEFINIZIONE DELLE STRATEGIE DI BASE
% ===================================

% --- Strategia Aggressiva ---
strategy_aggressive(M,S,MyShips) :-
    my_system(M),
    neutral_system(S),
    undirected_connected(M,S),
    ships(M,MyShips),
    MyShips > 10.  % Soglia bassa per attaccare

strategy_aggressive(M,S,MyShips) :-
    my_system(M),
    enemy_system(S),
    undirected_connected(M,S),
    ships(M,MyShips),
    ships(S,EnemyShips),
    MyShips > EnemyShips + 5.  % Minore vantaggio richiesto

% --- Strategia Difensiva ---
strategy_defensive(M,S,Ships) :-
    my_system(M),
    my_system(S),
    M != S,
    undirected_connected(M,S),
    border_system(S),  % Priorità ai sistemi di frontiera
    ships(M,MyShips),
    MyShips > 15,
    Ships = MyShips / 2.

strategy_defensive(M,S,EnemyShips) :-
    my_system(M),
    enemy_system(S),
    undirected_connected(M,S),
    ships(M,MyShips),
    ships(S,EnemyShips),
    Advantage = EnemyShips + (EnemyShips / 2),
    MyShips > Advantage.


% --- Strategia Tecnologica (Focus sui Sistemi Produttivi) ---
strategy_technological(M,S,MyShips) :-
    my_system(M),
    neutral_system(S),
    high_production_system(S),  % Priorità ai sistemi con alta produzione
    undirected_connected(M,S),
    ships(M,MyShips),
    MyShips > 20.

strategy_technological(M,S,MyShips) :-
    my_system(M),
    enemy_system(S),
    high_production_system(S),  % Priorità ai sistemi con alta produzione
    undirected_connected(M,S),
    ships(M,MyShips),
    MyShips > 20.

% --- Strategia Espansione (Conquistare Più Territori) ---
strategy_expansion(M,S,MyShips) :-
    my_system(M),
    neutral_system(S),
    undirected_connected(M,S),
    ships(M,MyShips),
    MyShips > 8.

% --- Strategia Consolidamento (Rinforza Sistemi Chiave) ---
% Calcola se MyShips è maggiore di 1.5 volte TargetShips (usando moltiplicazione intera per evitare float)
% Dichiarazione di dominio: My e Target provengono sempre da ships/2
dom_ships(My)   :- ships(_,My).
dom_ships(Target) :- ships(_,Target).


% Strategia di consolidamento
strategy_consolidation(M,S,Ships) :-
    my_system(M),
    my_system(S),
    M != S,
    undirected_connected(M,S),
    ships(M,MyShips),
    ships(S,TargetShips),
    MyShips*10 > TargetShips*15,
    ship_balancing(M,S,Ships).

% ===============================
% PREDICATI AUSILIARI
% ===============================

% Connessione non orientata
undirected_connected(M,S) :- connected(M,S).
undirected_connected(M,S) :- connected(S,M).

% Sistema di confine (vicino a nemici o neutrali)
border_system(S) :-
    my_system(S),
    undirected_connected(S,N),
    not my_system(N).

% Sistema centrale (non di confine)
central_system(S) :-
    my_system(S),
    not border_system(S).

% Sistema con alta produzione
high_production_system(S) :-
    system(S),
    production(S,P),
    P >= 5.

% Calcola bilanciamento ottimale di navi tra due sistemi
ship_balancing(M,S,Ships) :-
    my_system(M),
    my_system(S),
    ships(M,MyShips),
    ships(S,TargetShips),
    TotalShips = MyShips + TargetShips,
    IdealBalance = TotalShips / 2,
    Transfer = IdealBalance - TargetShips,
    Transfer > 0,
    Ships = Transfer.

% Definizioni di base per i sistemi
my_system(S)   :- system(S), owner(S,P), ai_player(P).
enemy_system(S):- system(S), owner(S,P), not ai_player(P), not neutral(S).
neutral_system(S):- neutral(S).

% ================================
% ANALISI DELLO STATO DEL GIOCO
% ================================

% Conta i sistemi posseduti dal giocatore IA
my_system_count(Count) :- Count = #count{S : my_system(S)}.

% Conta i sistemi nemici
enemy_system_count(Count) :- Count = #count{S : enemy_system(S)}.

% Conta i sistemi neutrali
neutral_system_count(Count) :- Count = #count{S : neutral_system(S)}.

% Conta le navi totali dell'IA
my_ships_total(Total) :- Total = #sum{Ships,S : my_system(S), ships(S,Ships)}.

% Conta le navi totali nemiche
enemy_ships_total(Total) :- Total = #sum{Ships,S : enemy_system(S), ships(S,Ships)}.

% Numero medio di navi nei sistemi dell'IA
my_ships_average(Avg) :-
    my_ships_total(Total),
    my_system_count(Count),
    Count > 0,
    Avg = Total / Count.

% ================================
% MEMORIA E CONFRONTO STATI
% ================================

% Questi fatti devono essere generati dal programma Java confrontando i file precedenti
% previous_my_system_count(N).

% ================================
% REGOLE DI SELEZIONE STRATEGIA
% ================================

% Confronto tra stato attuale e precedente
systems_gained :-
    my_system_count(Current),
    previous_my_system_count(Previous),
    Current > Previous.

systems_lost :-
    my_system_count(Current),
    previous_my_system_count(Previous),
    Current < Previous.

ships_increased :-
    my_ships_total(Current),
    previous_my_ships_total(Previous),
    Current > Previous.

ships_decreased :-
    my_ships_total(Current),
    previous_my_ships_total(Previous),
    Current < Previous.

enemy_weakened :-
    enemy_ships_total(Current),
    previous_enemy_ships_total(Previous),
    Current < Previous.

enemy_strengthened :-
    enemy_ships_total(Current),
    previous_enemy_ships_total(Previous),
    Current > Previous.

% Valutazione della situazione attuale
outnumbered :-
    my_ships_total(MyTotal),
    enemy_ships_total(EnemyTotal),
    MyTotal < EnemyTotal * 2.

superior :-
    my_ships_total(MyTotal),
    enemy_ships_total(EnemyTotal),
    MyTotal * 10 > EnemyTotal * 13.

neutral_opportunity :- neutral_system_count(Count), Count > 0.

% ================================
% SELEZIONE DELLA STRATEGIA
% ================================

use_aggressive :- superior, enemy_weakened.
use_aggressive :- superior, systems_gained.

% Scegli la strategia difensiva quando siamo in svantaggio o abbiamo perso terreno
use_defensive :- outnumbered, not superior.
use_defensive :- ships_decreased, not superior.
use_defensive :- systems_lost, not superior.


% Scegli la strategia tecnologica quando c'è equilibrio e opportunità di crescita
use_technological :-
    not outnumbered,
    not superior,
    neutral_opportunity,
    not systems_lost.

% Scegli la strategia di espansione quando abbiamo guadagnato navi ma non territori
use_expansion :-
    ships_increased,
    not systems_gained,
    neutral_opportunity.

% Scegli la strategia di consolidamento quando abbiamo guadagnato territori
use_consolidation :-
    systems_gained,
    my_system_count(Count),
    Count > 2.

% Strategia di default se nessuna condizione speciale è soddisfatta
use_expansion :- not use_aggressive, not use_technological,
                 not use_defensive, not use_consolidation.

% ================================
% APPLICAZIONE DELLA STRATEGIA
% ================================

% Applica la strategia selezionata
send_fleet(M,S,Ships) :- use_aggressive, strategy_aggressive(M,S,Ships).
send_fleet(M,S,Ships) :- use_defensive, strategy_defensive(M,S,Ships).
send_fleet(M,S,Ships) :- use_technological, strategy_technological(M,S,Ships).
send_fleet(M,S,Ships) :- use_expansion, strategy_expansion(M,S,Ships).
send_fleet(M,S,Ships) :- use_consolidation, strategy_consolidation(M,S,Ships).
