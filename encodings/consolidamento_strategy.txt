% Dominio delle distanze
distance(0..10).

% Distanza base: 0 per i sistemi di confine
dist_from_border(S, 0) :- border_system(S).

% Propagazione della distanza ai vicini
dist_from_border(S, D) :-
    my_system(S),
    undirected_connected(S, S1),
    dist_from_border(S1, D1),
    D = D1 + 1,
    distance(D),
    not smaller_distance_already_found(S, D).

% Supporto: verifica se c'è già una distanza minore
smaller_distance_already_found(S, D) :-
    dist_from_border(S, D2),
    distance(D2),
    distance(D),   % <-- vincola D per la safety
    D2 < D.

% Calcola il percorso ottimale verso il confine
next_step_to_border(From, To) :-
    my_system(From),
    my_system(To),
    undirected_connected(From, To),
    dist_from_border(From, DistFrom),
    dist_from_border(To, DistTo),
    DistFrom > DistTo,
    DistFrom > 0.

% Verifica se un sistema ha navi assegnate dal primo file
has_assignment(From) :- send_fleet(From, _, _).

available_for_consolidation(From, Available) :-
    my_system(From),
    ships(From, Total),
    not has_assignment(From),
    Total > 20,              % Soglia più bassa
    Available = Total - 5,   % Riserva più bassa
    Available > 10.          % Soglia minima più bassa

can_send_ships(From, Ships) :-
    available_for_consolidation(From, Available),
    Available >= 10,         % Soglia molto più bassa
    Ships = Available * 3 / 4,
    Ships >= 5.              % Minimo più basso

% Un sistema può consolidare
can_consolidate(From, To, Ships) :-
    next_step_to_border(From, To),
    can_send_ships(From, Ships).


% ===== PRIORITÀ DI CONSOLIDAMENTO =====
% Identifica sistemi con nemici vicini
has_enemy_neighbor(BorderSys) :-
    border_system(BorderSys),
    undirected_connected(BorderSys, EnemySys),
    enemy_system(EnemySys, _).

% Calcola il livello di minaccia
border_threat_level(BorderSys, high) :-
    border_system(BorderSys),
    undirected_connected(BorderSys, E1),
    enemy_system(E1, _),
    undirected_connected(BorderSys, E2),
    enemy_system(E2, _),
    E1 != E2.

border_threat_level(BorderSys, medium) :-
    border_system(BorderSys),
    has_enemy_neighbor(BorderSys),
    not border_threat_level(BorderSys, high).

border_threat_level(BorderSys, low) :-
    border_system(BorderSys),
    not has_enemy_neighbor(BorderSys).

% ===== SELEZIONE AZIONI DI CONSOLIDAMENTO =====
% Genera potenziali azioni di consolidamento
potential_consolidation(From, To, Ships) :- can_consolidate(From, To, Ships).

% ===== GESTIONE LIMITI PER DIFFICOLTÀ =====
max_actions(3) :- difficulty(medium).
max_actions(5) :- difficulty(hard).

% al massimo Max azioni di consolidamento
{ consolidation_action(From,To,Ships) : potential_consolidation(From,To,Ships) } <= Max :-max_actions(Max).

:- consolidation_action(From, To1, _), consolidation_action(From, To2, _), To1 != To2.
:- consolidation_action(From1, To, _), consolidation_action(From2, To, _), From1 != From2.


% ===== PREVENZIONE CONFLITTI =====
:- consolidation_action(From, To, _), send_fleet(From, To, _).


% ===== VINCOLI DI SICUREZZA RILASSATI =====
:- consolidation_action(From, _, Ships), ships(From, Total), Ships >= Total - 5.
:- consolidation_action(_, _, Ships), Ships <= 0.
:- consolidation_action(From, _, Ships), available_for_consolidation(From, Available), Ships > Available.
:- consolidation_action(_, To, _), not my_system(To).

% ===== WEAK CONSTRAINTS PER OTTIMIZZAZIONE =====

% ———————————————————————————————
% 1) Massimizzare il numero di consolidation action
% ogni potenziale non scelto costa 1 punto, PRIORITÀ 4 (più alto)
:~ potential_consolidation(From,To,Ships), not consolidation_action(From,To,Ships). [1@4,From]


% ———————————————————————————————
% 2) Premia consolidamenti con più navi
% peso inverso: azioni con poche navi costano di più, PRIORITÀ 3
% (sostituisci  max_ships  con un upper-bound sui Ships se necessario)
weight(From, To, W) :- consolidation_action(From, To, Ships), max_ships(M), W = M - Ships + 1.
:~ weight(From, To, W). [W@3, From, To]


% ———————————————————————————————
% 3) Premia sistemi con tante navi disponibili
% azioni che partono da fonti con poche navi costano di più, PRIORITÀ 2
max_available(A) :- A = #max{ Av : available_for_consolidation(_,Av) }.
:~ consolidation_action(From,_,_), available_for_consolidation(From,Av), max_available(A), W = (A - Av) + 1. [W@2]

% ———————————————————————————————
% 4) Penalizza consolidamento verso sistemi a basso/medio rischio
% PRIORITÀ 1 (più basso)
:~ consolidation_action(_,To,_), border_system(To), border_threat_level(To, low).    [30@1]
:~ consolidation_action(_,To,_), border_system(To), border_threat_level(To, medium). [10@1]

% ———————————————————————————————
% 5) Premia invii verso sistemi deboli (poche navi)
% PRIORITÀ 1 (stessa del rischio, o spostala a @0 se vuoi ancora più basso)
:~ consolidation_action(_,To,_), border_system(To), ships(To,S), W = 50 - S. [W@1]


consolidation_fleet(From, To, Ships) :- consolidation_action(From, To, Ships).
